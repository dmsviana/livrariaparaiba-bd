-- Selecionar todos os clientes que possuem telefone
SELECT *
FROM Pessoa p
INNER JOIN Cliente c ON p.id_pessoa = c.id_pessoa
WHERE p.telefone IS NOT NULL;

-- Selecionar todos os livros da editora "Editora A"
SELECT *
FROM Livro l
INNER JOIN Editora e ON l.id_editora = e.id_editora
WHERE e.nome = 'Editora A';

-- Selecionar todas as compras realizadas em "2020-01-01"
SELECT *
FROM Compra c
WHERE c.data = '2020-01-01';
